\begin{frame}[fragile]{SIMD related bugfixes}
 \begin{itemize}
     \item Fix simd arithmetic operators calling host-only functions on device
     \item Make masked version of simd reduce with default arguments unambiguous
     \item Fix segfaults when doing unaligned simd stores on AVX512
     \item Fix compile failures on AppleClang 14 and Clang 14 and 15 when using simd
 \end{itemize}
\end{frame}

\begin{frame}[fragile]{Other important bugfixes}
 \begin{itemize}
   \item Only use \texttt{\_\_atomic\_max/min\_fetch} with floating point types with LLVM $>$22.1
   \item Fix RangePolicy \texttt{HIP} performance regression with static batch size 1
   \item Fix bug in compiler detection for RISC-V with clang
   \item Fix thread-safety of \texttt{parallel\_scan} when resizing shared memory between kernels in \texttt{CUDA}
   \item Fix construction of an unmanaged subview from a \texttt{View}
   \item Fix using \texttt{Kokkos::UnorderedMap} as a set
 \end{itemize}
\end{frame}
