\begin{frame}[fragile]

  {\Huge Batched updates:}\\
  \vspace{10pt}

  Supporting general banded matrix and general matrix solvers
  \begin{itemize}
  \item Batched serial \texttt{lacgv}: Conjugates the length n complex vector
  \item Batched serial \texttt{syr}: Performs the rank 1 operation A:= alpha*x*x**T + A
  \item Batched serial \texttt{ger}: Performs the rank 1 operation A:= alpha*x*y**T + A
  \item Batched serial \texttt{gbtrf}: LU factorization of a general banded matrix A
  \item Batched serial \texttt{getrf}: LU factorization of a general matrix A
  \item Batched serial \texttt{getrs}: Solves a general system A * x = b factorized by \texttt{getrf}
  \end{itemize}  
\end{frame}
